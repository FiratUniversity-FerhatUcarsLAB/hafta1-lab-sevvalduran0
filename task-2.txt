FONKSIYON EnIyiHamleyiBul(tahta, oyuncu):
    enIyiSkor ← -∞
    enIyiHamle ← HİÇBİRİ

    HER hamle IN BosHucreler(tahta):
        tahta[hamle] ← oyuncu
        skor ← Minimax(tahta, 0, SAHTE, oyuncu)
        tahta[hamle] ← " "

        EĞER skor > enIyiSkor İSE
            enIyiSkor ← skor
            enIyiHamle ← hamle

    DÖNDÜR enIyiHamle


FONKSIYON Minimax(tahta, derinlik, maksimizeEdiyorMu, oyuncu):
    rakip ← EĞER oyuncu = "X" İSE "O" DEĞİLSE "X"

    EĞER KazandiMi(tahta, oyuncu) İSE
        DÖNDÜR +1
    EĞER KazandiMi(tahta, rakip) İSE
        DÖNDÜR -1
    EĞER BeraberlikMi(tahta) İSE
        DÖNDÜR 0

    EĞER maksimizeEdiyorMu İSE
        enIyiSkor ← -∞
        HER hamle IN BosHucreler(tahta):
            tahta[hamle] ← oyuncu
            skor ← Minimax(tahta, derinlik + 1, SAHTE, oyuncu)
            tahta[hamle] ← " "
            enIyiSkor ← MAX(enIyiSkor, skor)
        DÖNDÜR enIyiSkor

    DEĞİLSE
        enKotuSkor ← +∞
        HER hamle IN BosHucreler(tahta):
            tahta[hamle] ← rakip
            skor ← Minimax(tahta, derinlik + 1, DOĞRU, oyuncu)
            tahta[hamle] ← " "
            enKotuSkor ← MIN(enKotuSkor, skor)
        DÖNDÜR enKotuSkor
